  <script>
    const yesBtn=document.getElementById("yes");
    const noBtn=document.getElementById("no");
    const msg=document.getElementById("msg");
    const question=document.getElementById("question");
    const effects=document.querySelector(".effects");
    const falling=document.querySelector(".falling-hearts");

    function setDisabled(state=true){
      [yesBtn,noBtn].forEach(b=>b.disabled=state);
    }
    function showMessage(text,sub="",pulse=false){
      msg.innerHTML=text+(sub?`<br><small>${sub}</small>`:"");
      msg.classList.add("show");
      if(pulse) msg.classList.add("pulse");
    }
    function heartBurst(x,y,count=24){
      for(let i=0;i<count;i++){
        const h=document.createElement("span");
        h.textContent="❤";
        h.style.position="absolute";
        h.style.left=(x)+"px"; h.style.top=(y)+"px";
        h.style.fontSize=(Math.random()*16+10)+"px";
        const angle=Math.random()*Math.PI*2;
        const dist=Math.random()*120+40;
        const tx=Math.cos(angle)*dist;
        const ty=Math.sin(angle)*dist;
        h.style.transition="transform 1.2s ease,opacity 1.2s";
        effects.appendChild(h);
        requestAnimationFrame(()=>{
          h.style.transform=`translate(${tx}px,${ty}px) scale(.8)`;
          h.style.opacity="0";
        });
        setTimeout(()=>h.remove(),1200);
      }
    }
    yesBtn.addEventListener("click",()=>{
      setDisabled(true);
      showMessage("Benim en büyük cevabım daima sensin 💖","Kalbim sonsuza kadar seninle.",true);
      const r=yesBtn.getBoundingClientRect();
      heartBurst(r.left+r.width/2,r.top+r.height/2);
      question.textContent="Kalplerimiz birbirine kilitlendi!";
    });
    noBtn.addEventListener("click",()=>{
      yesBtn.textContent="Evet"; noBtn.textContent="Tabii ki Evet";
      setDisabled(true);
      setTimeout(()=>{
        showMessage("Aslında cevabın çoktan belli… ❤️","Senin kalbin bana ait.",true);
        question.textContent="Cevap kalbinden geldi!";
      },3000);
    });

    // Sürekli düşen kalpler
    function spawnFallingHeart(){
      const h=document.createElement("span");
      h.textContent="❤";
      h.style.left=Math.random()*100+"vw";
      h.style.fontSize=(Math.random()*12+10)+"px";
      h.style.animationDuration=(Math.random()*4+6)+"s";
      falling.appendChild(h);
      setTimeout(()=>h.remove(),10000);
    }
    setInterval(spawnFallingHeart,500);
  </script>
